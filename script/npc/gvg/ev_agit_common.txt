prontera.gat,152,208,0	script	GvG Control	111,{
	if ( !(getgmlevel(0)) ) close;
	mes "[GvG Control^ff0000(GM専用)^000000]";
	mes "攻城戦モード設定";
	next;
	switch ( select("開始","停止","キャンセル","デバッグ") ) {
	case 1:	announce "攻城戦開始",0;
		agitstart;
		break;
	case 2:	announce "攻城戦終了",0;
		agitend;
		break;
	case 4:	for(set $@ii,0;$@ii<7;set $@ii,$@ii+1){
		mes $@ii+","+$GvGDay[$@ii];
		}
		if(GvGTimeST != GvGTimeED) {
			mes "毎回　"+GvGTimeST+" 〜 "+GvGTimeED;
		} else {
			for(set $@ii,0;$@ii<7;set $@ii,$@ii+1){
			mes $@ii+","+$GvGST[$@ii]+" 〜 "+$GvGED[$@ii];
			}
		}
	}
	close;

//----------GVGの開始設定(Const.txtで設定可能)-------------
//
OnMinute00:
//24時終了の例外処理
	if(GetTime(3) == 0){
		set $@gved,GetTime(4)-1;
		if($@gved<0) set $@gved,6;
		if(GvGTimeST== GvGTimeED){
			if($GvGED[$@gved]==24)&&($GvGDay[$@gved]==1){
				if(!(($GvGDay[GetTime(4)]==1)&&($GvGST[GetTime(4)]==0))){
					goto LGvGEnd;
				} else end;
			}
		} else {
			if(GvGTimeED==24)&&($GvGDay[$@gved]==1){
				if(!(($GvGDay[GetTime(4)]==1)&&(GvGTimeST==0))){
					goto LGvGEnd;
				} else end;
			}
		}
	}

	if($GvGDay[GetTime(4)] != 1) end;

//毎回同じ時間
	if(GvGTimeST == GvGTimeED) goto LDayTimeCheck;
	if (GetTime(3) == GvGTimeST) goto LGvGStart;
	if (GetTime(3) == GvGTimeED) goto LGvGEnd;
	end;

//曜日ごと時間指定
LDayTimeCheck:
	if (GetTime(3) == $GvGST[GetTime(4)]) goto LGvGStart;
	if (GetTime(3) == $GvGED[GetTime(4)]) goto LGvGEnd;
	end;	
	
LGvGStart:
	announce "攻城戦開始",8;
	agitstart;
	end;
LGvGEnd:
	announce "攻城戦終了",8;
	agitend;
	end;
//----------時間中再起動時の再開処理(再起動1分後に判断)-----------
//曜日指定を配列に入れる処理をここでやっています。
//本当はconst.txtで配列が使えれば簡単なんだけど。定数の設定だからなぁ・・・
//GvGWeekDay 
//無し＝０、日＝１、月＝２、火＝４、水＝８、木＝１６、金＝３２、土＝６４
OnInit:		
	set $@GWI,GvGWeekDay;
	for(set $@ii,0;$@ii<7;set $@ii,$@ii+1){
		set $@GWI2,$@GWI/2;
		set $GvGDay[$@ii],($@GWI-$@GWI2*2);
		set $@GWI,$@GWI2;
	}
//各曜日の時間も配列へ。毎日同じ時間にやるなら処理しない。
	if(GvGTimeST != GvGTimeED) goto LInitEnd;
	set $GvGST[0],GvGTime0ST;
	set $GvGST[1],GvGTime1ST;
	set $GvGST[2],GvGTime2ST;
	set $GvGST[3],GvGTime3ST;
	set $GvGST[4],GvGTime4ST;
	set $GvGST[5],GvGTime5ST;
	set $GvGST[6],GvGTime6ST;
	set $GvGED[0],GvGTime0ED;
	set $GvGED[1],GvGTime1ED;
	set $GvGED[2],GvGTime2ED;
	set $GvGED[3],GvGTime3ED;
	set $GvGED[4],GvGTime4ED;
	set $GvGED[5],GvGTime5ED;
	set $GvGED[6],GvGTime6ED;
LInitEnd:
	initnpctimer;
	end;

OnTimer60000:
	if (GvGWeekDay == 0) goto LTimerEnd;
	
LNextTimeCheck:
	if($GvGDay[GetTime(4)] == 1){
		if (GvGTimeST != GvGTimeED) {
			if ((GetTime(3) >= GvGTimeST) && (GetTime(3) < GvGTimeED)){
				announce "攻城戦再開",8;
				agitstart;
			}
		} else {
			if((GetTime(3) >= $GvGST[GetTime(4)]) && (GetTime(3) < $GvGED[GetTime(4)])){
				announce "攻城戦再開",8;
				agitstart;
			}
		}
	}
LTimerEnd:
	stopnpctimer;
	end;
}

//宝箱発生処理
//callfunc "make_t_box","MAP",x0,y0,x1,y1,MobID1,MobID2;
function	script	make_t_box	{
	killmonster getarg(0),"TresureBox";
	set $@boxnum1,2+(getcastledata(getarg(0),2)+5)/10;
	set $@boxnum2,2+getcastledata(getarg(0),2)/10;
	if ($@boxnum1 > 12)	set $@boxnum1,12;
	if ($@boxnum2 > 12)	set $@boxnum2,12;
	areamonster getarg(0),getarg(1),getarg(2),getarg(3),getarg(4),"宝箱",getarg(5),$@boxnum1,"TresureBox";
	areamonster getarg(0),getarg(1),getarg(2),getarg(3),getarg(4),"宝箱",getarg(6),$@boxnum2,"TresureBox";
	SetCastleData getarg(0),2,GetCastleData(getarg(0),2)+GetCastleData(getarg(0),4);
	SetCastleData getarg(0),3,GetCastleData(getarg(0),3)+GetCastleData(getarg(0),5);
	SetCastleData getarg(0),4,0;//商業投資回数リセット
	SetCastleData getarg(0),5,0;//防衛投資回数リセット
	return;
}
